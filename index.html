1. Открой delivery.yandex.kz
2. Нажми F12 → Console → вставь код ниже → Enter
3. Скопируй результат и вставь в поле:

(function(){const rows=document.querySelectorAll("tr.bodyRow-PJWXzD");if(!rows.length){alert("Обнови страницу");return;}const fk=Object.keys(rows[0]).find(k=>k.startsWith("__reactFiber")||k.startsWith("__reactInternalInstance"));const o=[];rows.forEach(r=>{try{const c=r[fk]?.return?.memoizedProps?.claim;if(!c)return;const dest=(c.route||[]).filter(p=>p.type==="destination");if(!c.pricing?.total||!dest.length)return;o.push({date:c.createdAtDate,time:c.createdAtTime,total:Math.round(c.pricing.total),destinations_count:dest.length,cost_per_address:Math.round(c.pricing.total/dest.length),destinations:dest.map(d=>d.addressText)});}catch(e){}});navigator.clipboard.writeText(JSON.stringify(o,null,2)).then(()=>alert("✅ Скопировано! Вставь в поле на странице отчёта"));})();
